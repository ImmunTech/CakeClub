cake_data_raw <- read.csv("../data/2024_final_scores.tsv", sep = "\t", header = T)

    filter(Nationality_of_the_cake == "Swedish") %>%
    count(Cake)

    cake_data %>%
    filter(Nationality_of_the_cake != "Swedish") %>%

global_cakes <-

    global_cakes %>%
    filter(Score == max(Score)) %>%
    mutate(Score = round(Score, 1))
